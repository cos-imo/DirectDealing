Table User:
    User_id (Primary Key)
    First_Name
    Name
    Mail
    Password
    Photo_profil
    Wallet

Table Florain:
    florain_id (Primary Key)
    florain

Table Calendrier:
    Calendrier_id (Primary Key)

Table Creneau:
    Creneau_id (Primary Key)
    Calendrier_id (Foreign Key -> Calendrier)

Table Event:
    Event_id (Primary Key)
    type_id (Foreign Key -> type)
    Name
    preteur_id (Foreign Key -> User)
    acheteur_id (Foreign Key -> User)
    Recurrence
    Localisation
    Duree
    Couleur
    Prix_id (Foreign Key -> Florain)

Table Recurrence:
    Recurrence_id (Primary Key)

Table type:
    type_id (Primary Key)

Table Avis:
    Avis_id (Primary Key)
    Note
    Commentaire
    Auteur_id (Foreign Key -> User)
    Receveur_id (Foreign Key -> User)
    Date

Table Message:
    Message_id (Primary Key)
    Event_lie_id (Foreign Key -> Event)
    Sender_id (Foreign Key -> User)
    Receiver_id (Foreign Key -> User)
    Contenu
