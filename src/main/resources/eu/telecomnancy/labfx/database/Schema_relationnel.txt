--------------- CODE POUR CREATION DE LA BASE ---------------

-- USER --

CREATE TABLE User (
    User_id INTEGER PRIMARY KEY,
    First_Name TEXT,
    Name TEXT,
    Mail TEXT,
    Password TEXT,
    Photo_profil BLOB, -- Assumption: Image data is stored as BLOB
    Wallet INTEGER
);

-- EVENT --

CREATE TABLE Event (
    Event_id INTEGER PRIMARY KEY,
    type INTEGER, -- Assumption: 0 = service, 1 = prêt
    Name TEXT,
    preteur_id INTEGER REFERENCES User(User_id),
    acheteur_id INTEGER REFERENCES User(User_id),
    Recurrence INTEGER, -- Assumption: 0 = non, 1 = quotidien, 2 = mensuel, 3 = hebdomadaire
    Localisation TEXT,
    Duree REAL, -- en heure
    Couleur INTEGER, -- Assumption: 0 = Blanc(rien), 1 = Couleur1(prêteur dispo), 2 = Couleur2(prêteur avec créneau réservé), 3 = Couleur3(client avec créneau réservé)
    Prix INTEGER
);

-- AVIS --

CREATE TABLE Avis (
    Avis_id INTEGER PRIMARY KEY,
    Note INTEGER,
    Commentaire TEXT,
    Auteur_id INTEGER REFERENCES User(User_id),
    Receveur_id INTEGER REFERENCES User(User_id),
    Date DATE
);

-- MESSAGE -- 

CREATE TABLE Message (
    Message_id INTEGER PRIMARY KEY,
    Event_lie_id INTEGER REFERENCES Event(Event_id),
    Sender_id INTEGER REFERENCES User(User_id),
    Receiver_id INTEGER REFERENCES User(User_id),
    Contenu TEXT
);
